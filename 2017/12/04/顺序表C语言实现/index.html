顺序表C语言实现

2017-12-04 Data_Struture C/C++ 评论

今天学习了线性表，并手动实现了增删改查等操作，附上我的代码，最大的收获应该是对指针的理解和熟练程度又上一层楼了。
代码：

#include <stdio.h>
#include <stdlib.h>
#define OK 1
#define ERR 0
#define MAX_SIZE 20
//定义顺序存储结构
typedef int Elemtype;

typedef struct list{
 Elemtype data[MAX_SIZE];
 Elemtype Length;
}SeqList;
//初始化线性表①
SeqList *InitList1(){   //由于返回的是 线性表的指针数据类型  所以用 线性表数据类型定义一个指针函数 
    SeqList *L;    //定义一个线性表数据类型 指针
    L=(SeqList *)malloc(sizeof(SeqList));
    if(!L) return ERR;
     // 根据SeqList即线性表数据类型大小，通过malloc函数分配 合适 的逻辑上连续的地址空间，并强制转换为线性表数据类型（有可能会返回指针），这里返回的是一个可用空间首地址  
    L->Length=0; //L得到了返回可用空间的首地址，即成为了SqeList的引用，即可用 L->成员变量 的方式访问成员变量
    for(Elemtype i=0;i<MAX_SIZE;i++)
       L->data[i]=0;
    return L;
}

//初始化线性表② 
Elemtype InitList2(SeqList **p){
    *p=(SeqList *)malloc(sizeof(SeqList));
    if(!*p) return ERR;
    (*p)->Length=0;
    for(Elemtype i=0;i<MAX_SIZE;i++)
      (*p)->data[i]=0;
     return OK;
}


/**
 * 销毁线性表
*/
Elemtype DestoryList(SeqList *p){  //相当于 SeqList *p=地址; 传入函数内部  即p=地址
    free(p);
    p=NULL;
    return OK;
}

/**
 * 返回线性表L中的第i个位置的元素值给e  SearchList
*/
Elemtype SearchList(SeqList *L,Elemtype i,Elemtype e){
    if(i<1||i>L->Length||L->Length==0)
      return ERR;
      e=L->data[i-1];
      return  e;

}


/**
 * 在线性表L中查找与给定值e相等的元素，成功返回1，失败返回0 Locate
*/
Elemtype LocateList(SeqList *L,Elemtype e){
    if(L->Length==0)
    return ERR;
    for(Elemtype i=0;i<MAX_SIZE;i++){
        if(L->data[i]==e)
            return OK;
    }
     return ERR;
}


/**
 * 在线性表L中第i个位置之前插入新元素e  InsList
*/
Elemtype InsList(SeqList *L,Elemtype i,Elemtype e){
    if(L->Length==0||i<1||i>L->Length+1)
      return ERR;
    if(L->Length==MAX_SIZE)
    {
        printf("线性表已满\n");
        return ERR;
    }
   for(Elemtype j=L->Length-1;j>=i-1;j--){
            L->data[j+1]=L->data[j];
    }
     L->data[i-1]=e; 
     L->Length++;
    return OK;
}

/**
 * 删除线性表L中第i个位置元素，并使用e返回其值 DelList
*/
Elemtype DelList(SeqList *L,Elemtype i,Elemtype e){
    if(L->Length==0||i<1||i>L->Length)
       return ERR;

     e=L->data[i-1];
     if(i==L->Length)
           L->Length--;
    else for(Elemtype j=i-1;j<L->Length-1;j++)
             L->data[j]=L->data[j+1];
             L->Length--;
    return e;
}

/**
 * 线性表的输出 print_seqList
*/
void print_seqList(SeqList *L){
    for(Elemtype i=0;i<L->Length;i++)
    printf("%d  ",L->data[i]);
    printf("\n\n");
}
/**
 * 两个线性表连接 UniList
*/
Elemtype UniList1(SeqList *L1,SeqList *L2,SeqList **L3){
    Elemtype j=0,Length;
    Length=L1->Length+L2->Length;
   if(L1->Length==0||L2->Length==0||MAX_SIZE<Length)
      return ERR;

    for(Elemtype i=0;i<Length;i++){
          if(i<L1->Length){
             (*L3)->data[i]=L1->data[i];
             (*L3)->Length++;
          }
          else {              
              (*L3)->data[i]=L2->data[j];
              (*L3)->Length++;
              j++;
          }
    }
    return OK;
}

SeqList *UniList2(SeqList *L1,SeqList *L2,SeqList *L3){
    Elemtype j=0,Length;
    Length=L1->Length+L2->Length;
   if(L1->Length==0||L2->Length==0||MAX_SIZE<Length)
      return ERR;
    for(Elemtype i=0;i<Length;i++){
          if(i<L1->Length){
             L3->data[i]=L1->data[i];
             L3->Length++;
          }
          else {              
              L3->data[i]=L2->data[j];
              L3->Length++;
              j++;
          }
    }
    return L3;
}

    SeqList *L1,*L2,*L3;
    Elemtype init_flag=0,i,Destory_flag=0,e;
    //初始化一个线性表
    L1=InitList1();    //使用返回初始化好的线性表。
    InitList2(&L2);  // 传入需要初始化的线性表进行初始化。

    InitList2(&L3);
