快速排序法和插入排序法

使用的线性表作为数据载体，上代码

Liner_list.h 放在当前目录即可

#ifndef __LINER_LIST_H__
#define __LINER_LIST_H__
#include <stdio.h>
#include <stdlib.h>
#include <stdio.h>
#include <stdlib.h>
#define OK 1
#define ERR 0
#define MAX_SIZE 9
//定义顺序存储结构
typedef int Elemtype;

typedef struct list
{
    Elemtype data[MAX_SIZE];
    Elemtype Length;
} SeqList;
/**
 * 初始化顺序表① *InitList1()
*/
SeqList *InitList1()
{               //由于返回的是 顺序表的指针数据类型  所以用 顺序表数据类型定义一个指针函数
    SeqList *L; //定义一个顺序表数据类型 指针
    L = (SeqList *)malloc(sizeof(SeqList));
    // 根据SeqList即顺序表数据类型大小，通过malloc函数分配 合适 的逻辑上连续的地址空间，并强制转换为顺序表数据类型（有可能会返回指针），这里返回的是一个可用空间首地址
    L->Length = 0; //L得到了返回可用空间的首地址，即成为了SqeList的引用，即可用 L->成员变量 的方式访问成员变量
    for (Elemtype i = 0; i < MAX_SIZE; i++)
        L->data[i] = 0;
    return L;
}

/**
 * 初始化顺序表② InitList2()
*/
Elemtype InitList2(SeqList **p)
{
    *p = (SeqList *)malloc(sizeof(SeqList));
    (*p)->Length = 0;
    for (Elemtype i = 0; i < MAX_SIZE; i++) //初始化里面的值
        (*p)->data[i] = 0;
    return OK;
}

/** 
 * 销毁顺序表 DestoryList(SeqList *p)
*/
Elemtype DestoryList(SeqList *p)
{            //相当于 SeqList *p=地址; 传入函数内部  即p=地址
    free(p); //释放所分配的空间
    p = NULL;
    return OK;
}
/**
 * 销毁顺序表②  DestoryList1(SeqList *p)
*/
Elemtype DestoryList1(SeqList **p)
{             //使用二级指针释放
    free(*p); //释放所分配的空间
    *p = NULL;
    return OK;
}
/**
 * 返回顺序表L中的第i个位置的元素值给e  SearchList(SeqList *L,Elemtype i,Elemtype e)
*/
Elemtype SearchList(SeqList *L, Elemtype i, Elemtype e)
{                                                 //传引用的方式，顺序表最大的优点就是查找第i号元素的值的时候很方便
    if (i < 1 || i > L->Length || L->Length == 0) //对顺序表和传入位置进行合法性判断
        return ERR;
    e = L->data[i - 1];
    return e;
}

/**
 *  LocateList(SeqList *L,Elemtype e)
 * 在顺序表L中查找与给定值e相等的元素，成功返回1，失败返回0 Locate
*/
Elemtype LocateList(SeqList *L, Elemtype e)
{                       //通过对比单链表，顺序表可以提前知道自身的长度
    if (L->Length == 0) //对传入的顺序表进行判断
        return ERR;
    for (Elemtype i = 0; i < L->Length; i++)
    { //迭代找值
        if (L->data[i] == e)
            return OK;
    }
    return ERR;
}

/**
 * 在顺序表L中第i个位置之前插入新元素e  InsList(SeqList *L,Elemtype i,Elemtype e)
*/
Elemtype InsList(SeqList *L, Elemtype i, Elemtype e)
{ //传引用的放大
    if (L->Length == 0 || i < 1 || i > L->Length + 1)
        return ERR;
    if (L->Length == MAX_SIZE)
    {
        printf("顺序表已满\n");
        return ERR;
    } //进行插入之前先判断所传入顺序表的合法性，包括0<长度<100，插入位置的合法性：i>=1 && i<当前长度加1
    for (Elemtype j = L->Length - 1; j >= i - 1; j--)
    {                                //从最后面一个元素开始移动,直到移动到第i个位置，也就是存储位置i-1
        L->data[j + 1] = L->data[j]; //往后面赋值移动
    }
    L->data[i - 1] = e; //赋值
    L->Length++;        //长度加一
    return OK;
}

/**
 * 删除顺序表L中第i个位置元素，并使用e返回其值 DelList(SeqList *L,Elemtype i,Elemtype e)
*/
Elemtype DelList(SeqList *L, Elemtype i, Elemtype e)
{                                                 //分情况删除
    if (L->Length == 0 || i < 1 || i > L->Length) //对顺序表的合法性进行判断
        return ERR;
    e = L->data[i - 1]; //赋值给e
    if (i == L->Length)
        L->Length--;
    else
        for (Elemtype j = i - 1; j < L->Length - 1; j++) //循环前移,从被删位置开始，移动到最后一个元素的前一个位置就停止，即到了Length-1就停止
            L->data[j] = L->data[j + 1];
    L->Length--; //长度减一
    return e;
}

/**
 * 顺序表的输出 print_seqList(SeqList *L)
*/
void print_seqList(SeqList *L)
{
    for (Elemtype i = 0; i < L->Length; i++)
        printf("%d  ", L->data[i]);
    printf("\n\n");
}
/**
 * 两个顺序表连接① UniList1(SeqList *L1,SeqList *L2,SeqList **L3)
*/
Elemtype UniList1(SeqList *L1, SeqList *L2, SeqList **L3)
{
    Elemtype j = 0, Length; //定义计数器和总长度
    Length = L1->Length + L2->Length;
    if (L1->Length == 0 || L2->Length == 0 || MAX_SIZE < Length) //判断总长度有没有超过最大长度，以及两个顺序表的合法性
        return ERR;

    for (Elemtype i = 0; i < Length; i++)
    {
        if (i < L1->Length)
        {
            (*L3)->data[i] = L1->data[i];
            (*L3)->Length++;
        }
        else
        {
            (*L3)->data[i] = L2->data[j];
            (*L3)->Length++;
            j++;
        }
    }
    return OK;
}
/**
 * 两个顺序表连接② *UniList2(SeqList *L1,SeqList *L2,SeqList *L3)
*/
SeqList *UniList2(SeqList *L1, SeqList *L2, SeqList *L3)
{ //
    Elemtype j = 0, Length;
    Length = L1->Length + L2->Length;
    if (L1->Length == 0 || L2->Length == 0 || MAX_SIZE < Length)
        return ERR;
    for (Elemtype i = 0; i < Length; i++)
    {
        if (i < L1->Length)
        {
            L3->data[i] = L1->data[i];
            L3->Length++;
        }
        else
        {
            L3->data[i] = L2->data[j];
            L3->Length++;
            j++;
        }
    }
    return L3;
}

/**
 * 快速排序法　减少赋值版
*/
#include <stdio.h>
#include "Liner_list.h"      //顺序表
void InsertSort(SeqList **L) //插入排序法
{
    int i, j;
    for (i = 2; i <= (*L)->Length; i++)
    {
        if ((*L)->data[i - 1] > (*L)->data[i])
        {
            (*L)->data[0] = (*L)->data[i];
            for (j = i - 1; (*L)->data[j] > (*L)->data[0]; j--) //这句的意思是从第 i个　元素的钱一个元素开始，如果比“哨兵”元素即当前第i个元素大的，都向后移一位
            {                                                   //因为第i个之前，每次都遵循　大数插“我”右边，小数插“我”左边　操作，所以每一个“我”的左边都比“我”小,每一个右边都                //比“我”大,也就是说在第i个之前都是有序的。
                (*L)->data[j + 1] = (*L)->data[j];              //后移操作
            }
            (*L)->data[j + 1] = (*L)->data[0]; //第i个数　插入到适合的位置，循环里面多减了一次，这里要加一
        }
    }
}
int Partition(SeqList *L, int low, int high)
{
    //定义一个基准点partition
    int partition = L->data[low];
    L->data[0] = partition;
    while (low < high)
    {
        while (low < high && partition <= L->data[high]) //在最右边开始寻找比基准点小的值
            high--;                                      //前移，条件满足可以直到low==high
        L->data[low] = L->data[high];                    //因为此时的右边的值比基准点小,所以基准点变成了右边指向的那个值
        while (low < high && partition >= L->data[low])  //在最左边开始寻找比基准点大的值
            low++;                                       //后移，条件满足可以直到low==high
        L->data[high] = L->data[low];                    //因为左边的值比基准点大,所以基准点换成左边指向的值
    }
    L->data[low] = L->data[0];
    return low;
}

int QuickSort(SeqList **L, int start, int end) //　以顺序表为例
{
    if (!(*L))
        return ERR;
    int partition;
    if (start < end)
    {
        // printf("%d,%d\n",start,end);
        partition = Partition(*L, start, end);
        QuickSort(&*L, start, partition - 1);
        QuickSort(&*L, partition + 1, end);
    }

    return OK;
}

{

    SeqList *L1;
    Elemtype i;

    L1 = InitList1(); //使用返回初始化好的顺序表。

        L1->Length++;
    }

    for (i = 1; i <= L1->Length; i++)
        printf("%d  ", L1->data[i]);
    printf("\n\n");
    if (L1->Length > 10)
    {
        printf("顺序表长度大于10使用快速排序:\n");
        QuickSort(&L1, 1, L1->Length);
    }
    else 
    {
        printf("顺序表长度小于10使用插入排序:\n");
        InsertSort(&L1);
    }
    // print_seqList(L1);
    for (i = 1; i <= L1->Length; i++)
        printf("%d  ", L1->data[i]);
    printf("\n\n");
    DestoryList(L1);

    return 0;
}
